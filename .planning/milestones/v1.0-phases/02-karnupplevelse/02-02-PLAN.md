**1. Skapa FriendListViewModel.swift:**

```swift
@Observable
@MainActor
class FriendListViewModel {
    var favorites: [FriendWeather] = []
    var others: [FriendWeather] = []
    var isLoading = false
    var errorMessage: String?
    var showDemoBanner = false
    var refreshToken = UUID()  // för auto-refresh
}
```

Metoder:

`func load(uid: String, friendService: FriendService, weatherService: AppWeatherService) async`:
1. Sätt `isLoading = true`, defer `isLoading = false`
2. Hämta `friends = try await friendService.fetchFriends(uid: uid)`
3. Om `friends.isEmpty` OCH `UserDefaults.standard.bool(forKey: "hideDemoData") == false`:
   - Använd `DemoFriendService.demoFriends` istället
   - Sätt `showDemoBanner = true`
4. Annars: `showDemoBanner = friends.contains { $0.isDemo }`
5. Hämta väder parallellt med `withThrowingTaskGroup(of: FriendWeather?.self)`:
   - För varje friend: om `cityLatitude` och `cityLongitude` finns, anropa `weatherService.currentWeather(latitude:longitude:)` och skapa `FriendWeather(friend: friend, weather: weather)`
   - Returnera nil för friends utan koordinater
6. Sortera resultat `sorted { $0.temperatureCelsius > $1.temperatureCelsius }` (varmast först)
7. Filtrera: `favorites = sorted.filter { $0.friend.isFavorite }`, `others = sorted.filter { !$0.friend.isFavorite }`
8. Vid error: sätt `errorMessage = "Kunde inte hämta väderdata: \(error.localizedDescription)"`

`func toggleFavorite(uid: String, friend: Friend, friendService: FriendService) async`:
1. `do { try await friendService.toggleFavorite(uid: uid, friend: friend) }`
2. Flytta FriendWeather-objektet mellan favorites/others lokalt (uppdatera isFavorite-flaggan)
3. catch: sätt `errorMessage` (visa "Du har redan 6 favoriter..." om FriendServiceError.maxFavoritesReached)

`func removeDemoData(uid: String, friendService: FriendService) async`:
1. `try await friendService.removeDemoFriends(uid: uid)`
2. `UserDefaults.standard.set(true, forKey: "hideDemoData")`
3. `showDemoBanner = false`
4. `favorites = favorites.filter { !$0.friend.isDemo }`
5. `others = others.filter { !$0.friend.isDemo }`

`func refresh(uid: String, friendService: FriendService, weatherService: AppWeatherService) async`:
1. `await weatherService.clearCache()`
2. `await load(uid: uid, friendService: friendService, weatherService: weatherService)`

extension Color {
    static func temperatureColor(celsius: Double) -> Color {
        switch celsius {
        case ..<0:       return Color(red: 0.2, green: 0.4, blue: 1.0)   // isblå
        case 0..<10:     return Color(red: 0.4, green: 0.6, blue: 0.9)   // kylig blå
        case 10..<20:    return Color(red: 0.5, green: 0.7, blue: 0.5)   // neutral grön
        case 20..<28:    return Color(red: 1.0, green: 0.6, blue: 0.2)   // varm orange
        default:         return Color(red: 0.9, green: 0.2, blue: 0.2)   // het röd
        }
    }
}
